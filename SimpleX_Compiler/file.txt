    method int mul(int x, int y) 
    {
        int result;
        int counter;

        let result = 0;
        let counter = y;
        while (counter != 0)
        {
            let result = result + x;
            let counter = counter - 1;
        }
              
        return result;
    }

    method int max(int x, int y) 
    {
        int result;
       
        let result = x;
        if (y > x)
        {
            let result = y;
        }
              
        return result;
    }

     method int min(int x, int y) 
    {
        int result;
       
        let result = x;
        if (y < x)
        {
            let result = y;
        }
              
        return result;
    }
}
