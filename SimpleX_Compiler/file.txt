        int dist;
        let px1 = 0;

        let px2 = 7;

    method int distance(int x1, int y1, int x2, int y2) 
    {
        int distX, distY;
    
        let distX = x2 - x1;
        let distY = y2 - y1;
       
        return (Math.sqrt(Math.mul(distX, distX) + Math.mul(distY, distY)));
    }

    method int add1(int x)
    {
        int res;

        let res = res + 1;

        return res;
    }
      
    method int mul(int x, int y) 
    {
        int result;
        int counter;

        let result = 0;
        let counter = y;
        while (counter != 0)
        {
            let result = result + x;
            let counter = counter - 1;
        }
              
        return result;
    }

    method int max(int x, int y) 
    {
        int result;
       
        let result = x;
        if (y > x)
        {
            let result = y;
        }
              
        return result;
    }

    method int min(int x, int y) 
    {
        int result;
       
        let result = x;
        if (y < x)
        {
            let result = y;
        }
              
        return result;
    }


    method int sqrt(int x) 
    {
        int N;
        int remainder;
        int result;
       
        let remainder = x;
        let N = 1;
        let result = 0;
        while (remainder >= N)
        {
            let remainder = remainder - N;
            let N = N + 2;
            let result = result + 1; 
        }
              
        return result;
    }

    method int div(int x, int y) 
    {
        int remainder;
        int result;
        int N;

        let remainder = x;
        let N = 0;
        let result = 0;
        while (remainder >= y)
        {
            let remainder = remainder - y;
            let N = N + 1;
            let result = result + 1;
        }
              
        return result;
    }
}
